  def initialize( xd=[], to=nil, tp=nil )
    @self = XmlListDelegate.new(self)
    case xd
    when []
      @list = []
      @target_object = to
      @target_property = tp
    when Xml
      @list = [xd]
      @target_object = xd.self.parent
      @target_property = xd.self.name
    else
      xd = REXML::Document.new(%{<_>#{xd}</_>}).root
      a = []; xd.each{ |n| a << Xml.new(n) }
      @list = a
      @target_object = nil
      @target_property = nil
    end
  end
